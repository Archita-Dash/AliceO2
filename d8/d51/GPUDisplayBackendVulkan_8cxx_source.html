111 unsigned int GPUDisplayBackendVulkan::drawVertices(const vboList& v, const drawType tt)
112 {
113  auto first = std::get<0>(v);
114  auto count = std::get<1>(v);
115  auto iSlice = std::get<2>(v);
116  if (count == 0) {
117  return 0;
118  }
119  if (mCommandBufferUpToDate[mImageIndex]) {
120  return count;
121  }
122 
123  vkCmdBindPipeline(mCommandBuffers[mImageIndex], VK_PIPELINE_BIND_POINT_GRAPHICS, mPipelines[tt]);
124  if (mDisplay->cfgR().useGLIndirectDraw) {
125  vkCmdDrawIndirect(mCommandBuffers[mImageIndex], mIndirectCommandBuffer[0].buffer, (mIndirectSliceOffset[iSlice] + first) * sizeof(DrawArraysIndirectCommand), count, sizeof(DrawArraysIndirectCommand));
126  } else {
127  for (unsigned int k = 0; k < count; k++) {
128  vkCmdDraw(mCommandBuffers[mImageIndex], mDisplay->vertexBufferCount()[iSlice][first + k], 1, mDisplay->vertexBufferStart()[iSlice][first + k], 0);
129  }
130  }
131 
132  return count;
133 }

154 static int checkValidationLayerSupport(const std::vector<const char*>& validationLayers)
155 {
156  uint32_t layerCount;
157  vkEnumerateInstanceLayerProperties(&layerCount, nullptr);
158  std::vector<VkLayerProperties> availableLayers(layerCount);
159  vkEnumerateInstanceLayerProperties(&layerCount, availableLayers.data());
160  for (const char* layerName : validationLayers) {
161  bool layerFound = false;
162 
163  for (const auto& layerProperties : availableLayers) {
164  if (strcmp(layerName, layerProperties.layerName) == 0) {
165  layerFound = true;
166  break;
167  }
168  }
169 
170  if (!layerFound) {
171  return 1;
172  }
173  }
174  return 0;

310 void GPUDisplayBackendVulkan::transitionImageLayout(VkImage image, VkFormat format, VkImageLayout oldLayout, VkImageLayout newLayout)
311 {
312  VkCommandBuffer commandBuffer = getSingleTimeCommandBuffer();
313  VkImageMemoryBarrier barrier{};
314  barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
315  barrier.oldLayout = oldLayout;
316  barrier.newLayout = newLayout;
317  barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
318  barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
319  barrier.image = image;
320  barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
321  barrier.subresourceRange.baseMipLevel = 0;
322  barrier.subresourceRange.levelCount = 1;
323  barrier.subresourceRange.baseArrayLayer = 0;
324  barrier.subresourceRange.layerCount = 1;
325  VkPipelineStageFlags sourceStage;
326  VkPipelineStageFlags destinationStage;
327  if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED && newLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
328  barrier.srcAccessMask = 0;
329  barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
330  sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
331  destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
332  } else if (oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL && newLayout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
333  barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
334  barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
335  sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
336  destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
337  } else {
338  throw std::invalid_argument("unsupported layout transition!");
339  }
340 
341  vkCmdPipelineBarrier(commandBuffer, sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
342  submitSingleTimeCommandBuffer(commandBuffer);
343 }

432 double GPUDisplayBackendVulkan::checkDevice(VkPhysicalDevice device, const std::vector<const char*>& reqDeviceExtensions)
433 {
434  double score = -1.;
435  VkPhysicalDeviceProperties deviceProperties;
436  vkGetPhysicalDeviceProperties(device, &deviceProperties);
437  VkPhysicalDeviceFeatures deviceFeatures;
438  vkGetPhysicalDeviceFeatures(device, &deviceFeatures);
439  VkPhysicalDeviceMemoryProperties memoryProperties;
440  vkGetPhysicalDeviceMemoryProperties(device, &memoryProperties);
441  if (!deviceFeatures.geometryShader || !deviceFeatures.wideLines || !deviceFeatures.largePoints) {
442  return (-1);
443  }
444 
445  uint32_t queueFamilyCount = 0;
446  vkGetPhysicalDeviceQueueFamilyProperties(device, &queueFamilyCount, nullptr);
447  std::vector<VkQueueFamilyProperties> queueFamilies(queueFamilyCount);
448  vkGetPhysicalDeviceQueueFamilyProperties(device, &queueFamilyCount, queueFamilies.data());
449  bool found = false;
450  for (unsigned int i = 0; i < queueFamilies.size(); i++) {
451  if (!(queueFamilies[i].queueFlags & VK_QUEUE_GRAPHICS_BIT)) {
452  return (-1);
453  }
454  VkBool32 presentSupport = false;
455  vkGetPhysicalDeviceSurfaceSupportKHR(device, i, mSurface, &presentSupport);
456  if (!presentSupport) {
457  return (-1);
458  }
459  mQueueFamilyIndices->graphicsFamily = i;
460  found = true;
461  break;
462  }
463  if (!found) {
464  GPUInfo("%s ignored due to missing queue properties", deviceProperties.deviceName);
465  return (-1);
466  }
467 
468  uint32_t deviceExtensionCount;
469  vkEnumerateDeviceExtensionProperties(device, nullptr, &deviceExtensionCount, nullptr);
470  std::vector<VkExtensionProperties> availableExtensions(deviceExtensionCount);
471  vkEnumerateDeviceExtensionProperties(device, nullptr, &deviceExtensionCount, availableExtensions.data());
472  unsigned int extensionsFound = 0;
473  for (unsigned int i = 0; i < reqDeviceExtensions.size(); i++) {
474  for (unsigned int j = 0; j < availableExtensions.size(); j++) {
475  if (strcmp(reqDeviceExtensions[i], availableExtensions[j].extensionName) == 0) {
476  extensionsFound++;
477  break;
478  }
479  }
480  }
481  if (extensionsFound < reqDeviceExtensions.size()) {
482  GPUInfo("%s ignored due to missing extensions", deviceProperties.deviceName);
483  return (-1);
484  }
485 
486  updateSwapChainDetails(device);
487  if (mSwapChainDetails->formats.empty() || mSwapChainDetails->presentModes.empty()) {
488  GPUInfo("%s ignored due to incompatible swap chain", deviceProperties.deviceName);
489  return (-1);
490  }
491 
492  score = 1;
493  if (deviceProperties.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) {
494  score += 1e12;
495  } else if (deviceProperties.deviceType == VK_PHYSICAL_DEVICE_TYPE_INTEGRATED_GPU) {
496  score += 1e11;
497  }
498 
499  for (unsigned int i = 0; i < memoryProperties.memoryHeapCount; i++) {
500  if (memoryProperties.memoryHeaps[i].flags & VK_MEMORY_HEAP_DEVICE_LOCAL_BIT) {
501  score += memoryProperties.memoryHeaps[i].size;
502  }
503  }
504 
505  return score;
506 }

511  appInfo.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
512  appInfo.pApplicationName = "Hello Triangle";
513  appInfo.applicationVersion = VK_MAKE_VERSION(1, 0, 0);

516  appInfo.apiVersion = VK_API_VERSION_1_0;
517

520  instanceCreateInfo.pApplicationInfo = &appInfo;
521

523  uint32_t frontendExtensionCount = mDisplay->frontend()->getReqVulkanExtensions(frontendExtensions);
524  std::vector<const char*> reqInstanceExtensions(frontendExtensions, frontendExtensions + frontendExtensionCount);
525 
526  const std::vector<const char*> reqValidationLayers = {
527  "VK_LAYER_KHRONOS_validation"};
528  auto debugCallback = [](VkDebugUtilsMessageSeverityFlagBitsEXT messageSeverity, VkDebugUtilsMessageTypeFlagsEXT messageType, const VkDebugUtilsMessengerCallbackDataEXT* pCallbackData, void* pUserData) -> VkBool32 {
529  switch (messageSeverity) {
530  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
531  // GPUInfo("%s", pCallbackData->pMessage);
532  break;
533  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
534  GPUWarning("%s", pCallbackData->pMessage);
535  break;
536  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
537  GPUError("%s", pCallbackData->pMessage);
538  break;
539  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
540  default:
541  GPUInfo("%s", pCallbackData->pMessage);
542  break;
543  }
544  return VK_FALSE;
545  };
546  if (mEnableValidationLayers) {
547  if (checkValidationLayerSupport(reqValidationLayers)) {
548  throw std::runtime_error("Requested validation layer support not available");
549  }
550  reqInstanceExtensions.push_back(VK_EXT_DEBUG_UTILS_EXTENSION_NAME);
551  instanceCreateInfo.enabledLayerCount = static_cast<uint32_t>(reqValidationLayers.size());
